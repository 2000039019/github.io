      #include<bits/stdc++.h>
      using namespace std;
      bool check(string s,string s1){
          if(s.size()!=s1.size())return false;
          for(int i=0;i<s.size();i++){
              if(s[i]!=s1[i])return false;
          }
          return true;
      }
      
      int main(){
         // write your code here
         int n;
         cin>>n;
         while(n--){
             string s,s1;
             cin>>s>>s1;
             sort(s.begin(),s.end());
             sort(s1.begin(),s1.end());
             if(check(s,s1))cout<<"True"<<"\n";
             else cout<<"False"<<"\n";
         }
          return 0;
      }
