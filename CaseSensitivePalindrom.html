C++ Code Display

    
      #include<bits/stdc++.h>
      using namespace std;

      int main(){
         // write your code here
         int t;
         cin>>t;
         while(t--){
             string s;
             cin>>s;
             transform(s.begin(), s.end(), s.begin(), ::tolower);

             bool flag=false;
             int n=s.size();
             for(int i=0;i<n/2;i++){
                 if(s[i]!=s[n-1-i]){
                     flag=true;
                 }
             }
             if(flag){
                 cout<<"No"<<"\n";
             }
             else{
                 cout<<"Yes"<<"\n";
             }

         }
          return 0;
      }
